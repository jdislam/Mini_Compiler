int main() {
  char x = 'b';
  char y = 'b';
  string a = "hello";
  string b = "world";
  string c = "hello";

  // equality on char / string allowed
  print(x == y);   // 1
  print(a == b);   // 0
  print(a == c);   // 1

  // numeric + boolean
  int n = 3;
  int m = 6;
  bool ok = (n * 2 == m) && (n < m);
  print(ok);       // 1

  return;
}
